在我们生活中随处可见二维码，我们在零知板上使用软件库来生成并显示二维码。
硬件：零知-标准板、OLED（最好用像素更大的LCD）
连线：连线请参照OLED模块使用教程进行。

程序源码如下：
  1. /**
  2. *  QRCode 零知板-二维码生成与显示
  3. * 2018年7月31日16:23:41
  4. *        by 零知实验室
  5. *
  6. */

  8. #include <qrcode.h>
  9. #include <Adafruit_SSD1306.h>

  11. #define OLED_RESET 4
  12. Adafruit_SSD1306 display(OLED_RESET);

  14. void setup() {
  15.     Serial.begin(9600);
  16.         
  17.         display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  18.         display.clearDisplay();

  20.     // Start time
  21.     uint32_t dt = millis();
  22.   
  23.     // 根据字符串生成二维码
  24.     QRCode qrcode;
  25.     uint8_t qrcodeData[qrcode_getBufferSize(3)];
  26.     qrcode_initText(&qrcode, qrcodeData, 3, 0, "http://www.lingzhilab.com");
  27.   
  28.     // Delta time
  29.     dt = millis() - dt;
  30.     Serial.print("QR Code Generation Time: ");
  31.     Serial.print(dt);
  32.     Serial.print("\n");

  34.     // Top quiet zone
  35.     Serial.print("\n\n\n\n");
  36.         

  38.         //显示到OLED上
  39.     for (uint8_t y = 0; y < qrcode.size; y++) {

  41.         // Left quiet zone
  42.         Serial.print("        ");

  44.         // Each horizontal module
  45.         for (uint8_t x = 0; x < qrcode.size; x++) {

  47.             // Print each module (UTF-8 \u2588 is a solid block)
  48.             //Serial.print(qrcode_getModule(&qrcode, x, y) ? "\u2588\u2588": "  ");
  49.                         
  50.                         if(qrcode_getModule(&qrcode,x,y)){
  51.                                 display.drawPixel(x, y, WHITE);
  52.                         }else{
  53.                                 display.drawPixel(x, y, BLACK);
  54.                         }

  56.         }

  58.         Serial.print("\n");
  59.     }
  60.         
  61.         display.display();

  63.     // Bottom quiet zone
  64.     Serial.print("\n\n\n\n");
  65. }

  67. void loop() {

  69. }
